「力扣」第 5 题：最长回文子串

  • 链接

题目描述

给你一个字符串 s，找到 s 中最长的回文子串。

 

示例 1：
输入：s = "babad"
输出："bab"
解释："aba" 同样是符合题意的答案。

示例 2：
输入：s = "cbbd"
输出："bb"

 

提示：

  • 1 <= s.length <= 1000
  • s 仅由数字和英文字母组成

 

 

参考代码

public class Solution {
    public String longestPalindrome(String s) {
        // 由于 s.charAt(i) 操作会去检查下标是否越界，因此
        // 在 Java 里，将字符串转换成字符数组是常见的操作
        char[] chars = s.toCharArray();

        // 添加特殊字符 '#' 的字符数组
        char[] res = new char[(s.length() << 1) + 1];
        for (int i = 0; i < res.length; i++) {
            res[i] = ((i & 1) == 0) ? '#' : chars[i >> 1];
        }

        // 臂长：表示中心扩展算法向外扩展的长度。
        // 如果一个位置的最大回文字符串长度为 2 * length + 1 ，其臂长为 length。
        // p[i] 表示以 res[i] 为中心的臂长
        int[] p = new int[res.length];

        // 某个最大回文字符串延伸到的最右边下标
        int maxRight = 0;
        // maxRight 所属回文串中心下标
        int maxCenter = 0;

        // 记录遍历过的最大回文字符串中心下标
        int resCenter = 0;
        // 记录遍历过的最大臂长
        int resLen = 0;

        for (int i = 0; i < res.length; i++) {
            if (i < maxRight) {
                int j = (maxCenter << 1) - i;
                // 情况一：j 的臂长没有超出范围 [maxLeft, maxRight]，直接让 p[i] = p[j]
                if (p[j] < maxRight - i) {
                    p[i] = p[j];
                } else {
                    // 情况二：j 的臂长已经超出了范围 [maxLeft, maxRight]，
                    // 我们只能确定 p[i] 的最小值是 maxRight - i
                    p[i] = maxRight - i;
                    p[i] = expand(res, i - p[i], i + p[i]);
                }
            } else {
                // 情况三：i 超出了范围 [maxLeft, maxRight]，没法利用之前的已知数据
                p[i] = expand(res, i, i);
            }
            // 如果右边界 i + p[i] 超过 maxRight，那么就更新 maxRight 和 maxCenter
            if (i + p[i] > maxRight) {
                maxRight = i + p[i];
                maxCenter = i;
            }
            // 记录最长回文串的半径和中心位置
            if (p[i] > resLen) {
                resLen = p[i];
                resCenter = i;
            }
        }

        // 计算最长回文串的长度和开始的位置
        int start = (resCenter - resLen) >> 1;
        // 截取最长回文子串
        return s.substring(start, start + resLen);
    }

    private int expand(char[] res, int left, int right) {
        do {
            left--;
            right++;
        } while (left >= 0 && right < res.length && res[left] == res[right]);
        return (right - left - 2) >>> 1;
    }
}
